    def get_cls(cls, name):
        """
        Get a cutting strategy by name

        Returns:
            GcodeFormatter object
        """
        name = name.lower()
        
        if name == "default":
            return cls.default

        if name == "debug":
            return cls.debug

        for f in cls.formatters:
            f_name = f.__name__.lower()
            if f_name == name:
                return f

        logging.error("ERROR: GCODE FORMATTER NAME INCORRECT, FALLING BACK TO DEFAULT")
        return cls.default
